void decrypt()
{
    system("cls");
    TEM();

    char s[100], p[100], hex[20], x, y;
	int n=0, b=1, a=1, c, size, i, j;

	printf("  Enter a string: ");
	scanf(" %[^\n]",p);
	printf("  Enter your code: ");
	scanf(" %s",hex);

	size=strlen(hex);
	for(i=size-1; i>=0; i--)
    {
        if(hex[i] >='0' && hex[i]<= '9'){
            n+=(hex[i]-48)*b;
            b*= 16;
        }else if(hex[i]>='a' && hex[i]<='f'){
            n+=(hex[i]-87)*b;
            b*= 16;
        }
    }
    n/=((n%1000)*1000);
    n--;

    c=strlen(p);
    while(a<4)
    {
        for(i=0; i<c; i++)
        {
            s[i]=p[c-1-i];
        }

        for(i=0; i<c; i++)
        {
            char f=s[c-1-i];
            x=switch_box2(f);
            if(f<'a' || f>'z'){
                p[c-1-i]=s[c-1-i];
            }else if(x-(n%26)<97){
                p[c-1-i]=123-(97-(x-(n%26)));
            }else{
                p[c-1-i]=x-(n%26);
            }
            n--;
        }
        a++;
    }

    printf("\n  \033[1;32m%s\033[0m\n ",p);

    getch();
}

char switch_box1(char a, int n)
{
    char ff=a, nn=a;
    int mm=ff;

    if(ff>='A' && ff<='Z')
        ff-=32;

    if(mm+(n%26)>122){
        ff=96+(mm+(n%26))-122;
    }else{
        ff+=(n%26);
    }

    switch(ff){
        case 'a': return 'n';
        case 'b': return 'y';
        case 'c': return 'i';
        case 'd': return 'z';
        case 'e': return 'a';
        case 'f': return 'h';
        case 'g': return 'w';
        case 'h': return 'o';
        case 'i': return 'b';
        case 'j': return 'j';
        case 'k': return 'm';
        case 'l': return 'c';
        case 'm': return 's';
        case 'n': return 'x';
        case 'o': return 'd';
        case 'p': return 'v';
        case 'q': return 'p';
        case 'r': return 'e';
        case 's': return 'k';
        case 't': return 'q';
        case 'u': return 'f';
        case 'v': return 'u';
        case 'w': return 'r';
        case 'x': return 'g';
        case 'y': return 'l';
        case 'z': return 't';
        default: return nn;
    }
}
